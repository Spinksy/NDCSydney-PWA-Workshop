this.dbStore = new Store();

//add this function to the class
 _saveToStore(topic) {
    return navigator.serviceWorker.getRegistration().then(reg => {
      if ("sync" in reg) {
        this.dbStore
          .outbox('readwrite')
          .then(outbox => {
            return outbox.put(topic);
          })
          .then(() => {
            return reg.sync.register('outbox');
          });
      }
    });
  }
